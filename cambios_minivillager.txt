private void applyEffects(LivingEntity villager) {
    // Siempre añadir efecto de brillo para hacerlo más visible
    villager.addPotionEffect(new PotionEffect(PotionEffectType.GLOWING, Integer.MAX_VALUE, 0, false, false));

    // Efectos adicionales según velocidad
    if (speedLevel >= 3) {
        // Más velocidad, más partículas
        villager.getWorld().spawnParticle(
            org.bukkit.Particle.VILLAGER_HAPPY,
            villager.getLocation().add(0, 0.5, 0),
            5, 0.2, 0.2, 0.2, 0
        );
    }
}
